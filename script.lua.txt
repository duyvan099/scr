local function ForcePart(v)
    if v:IsA("BasePart") and not v.Anchored and not v.Parent:FindFirstChildOfClass("Humanoid") and not v.Parent:FindFirstChild("Head") and v.Name ~= "Handle" then
        for _, x in ipairs(v:GetChildren()) do
            if x:IsA("BodyMover") or x:IsA("RocketPropulsion") then x:Destroy() end
        end
        if v:FindFirstChild("Attachment") then v:FindFirstChild("Attachment"):Destroy() end
        if v:FindFirstChild("AlignPosition") then v:FindFirstChild("AlignPosition"):Destroy() end
        if v:FindFirstChild("Torque") then v:FindFirstChild("Torque"):Destroy() end
        
        v.CanCollide = false
        local Torque = Instance.new("Torque", v)
        Torque.Torque = Vector3.new(100000, 100000, 100000)
        local AlignPosition = Instance.new("AlignPosition", v)
        local Attachment2 = Instance.new("Attachment", v)
        Torque.Attachment0 = Attachment2
        AlignPosition.MaxForce = math.huge
        AlignPosition.MaxVelocity = math.huge
        AlignPosition.Responsiveness = 200
        AlignPosition.Attachment0 = Attachment2
        AlignPosition.Attachment1 = Attachment1
    end
end

-- [ HỆ THỐNG ESP CŨ ]
local function CreateESP(player)
    local Objects = {
        Box = Drawing.new("Square"), Tracer = Drawing.new("Line"),
        HealthBar = Drawing.new("Line"), NameTag = Drawing.new("Text"), TeleBtn = Drawing.new("Text")
    }
    Objects.Box.Color = Settings.WhiteColor; Objects.Tracer.Color = Settings.WhiteColor
    Objects.NameTag.Color = Settings.WhiteColor; Objects.HealthBar.Color = Color3.fromRGB(0, 255, 0)
    Objects.TeleBtn.Color = Color3.fromRGB(0, 255, 255)
    Objects.Box.Thickness = 1; Objects.Box.Filled = false; Objects.Tracer.Thickness = 1
    Objects.HealthBar.Thickness = 2
    Objects.NameTag.Center = true; Objects.NameTag.Outline = true; Objects.NameTag.Size = 16
    Objects.TeleBtn.Text = "[ TELE ]"; Objects.TeleBtn.Size = 18; Objects.TeleBtn.Center = true; Objects.TeleBtn.Outline = true

    local Connection
    Connection = RunService.RenderStepped:Connect(function()
        if not player or not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
            for _, obj in pairs(Objects) do obj.Visible = false end
            if not player then Connection:Disconnect() end
            return
        end
        local RootPart = player.Character.HumanoidRootPart
        local Pos, OnScreen = Camera:WorldToViewportPoint(RootPart.Position)
        if OnScreen then
            local TopPos = Camera:WorldToViewportPoint(RootPart.Position + Vector3.new(0, 3, 0))
            local BottomPos = Camera:WorldToViewportPoint(RootPart.Position - Vector3.new(0, 3.5, 0))
            local SizeY = math.abs(TopPos.Y - BottomPos.Y)
            local SizeX = math.abs(SizeY / 2)
            local BoxPos = Vector2.new(Pos.X - SizeX / 2, Pos.Y - SizeY / 2)
            Objects.Box.Visible = Settings.Box
            Objects.Box.Size = Vector2.new(SizeX, SizeY); Objects.Box.Position = BoxPos
            Objects.Tracer.Visible = Settings.Line
            Objects.Tracer.From = Vector2.new(Camera.ViewportSize.X / 2, 0); Objects.Tracer.To = Vector2.new(Pos.X, BoxPos.Y)
            Objects.HealthBar.Visible = Settings.Health
            local hum = player.Character:FindFirstChild("Humanoid")
            if hum then
                local hPct = hum.Health / hum.MaxHealth
                Objects.HealthBar.From = Vector2.new(BoxPos.X - 5, BoxPos.Y + SizeY)
                Objects.HealthBar.To = Vector2.new(BoxPos.X - 5, BoxPos.Y + SizeY - (SizeY * hPct))
                Objects.HealthBar.Color = Color3.fromHSV(hPct * 0.3, 1, 1)
            end
            Objects.NameTag.Visible = Settings.Name
            local dist = math.floor((LocalPlayer.Character.HumanoidRootPart.Position - RootPart.Position).Magnitude)
            Objects.NameTag.Text = player.Name .. " [" .. dist .. "m]"
            Objects.NameTag.Position = Vector2.new(Pos.X, BoxPos.Y - 20)
            Objects.TeleBtn.Visible = Settings.Teleport
            Objects.TeleBtn.Position = Vector2.new(Pos.X, BoxPos.Y - 40)
            if Settings.Teleport and UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton1) then
                if (UserInputService:GetMouseLocation() - Objects.TeleBtn.Position).Magnitude < 30 then
                    LocalPlayer.Character.HumanoidRootPart.CFrame = RootPart.CFrame * CFrame.new(0, 0, 3)
                end
            end
        else
            for _, obj in pairs(Objects) do obj.Visible = false end
        end
    end)
end

for _, p in pairs(Players:GetPlayers()) do if p ~= LocalPlayer then CreateESP(p) end end
Players.PlayerAdded:Connect(function(p) if p ~= LocalPlayer then CreateESP(p) end end)

-- Logic Bring Parts (Mới)
local function getPlayer(name)
    local lowerName = string.lower(name)
    for _, p in pairs(Players:GetPlayers()) do
        if string.find(string.lower(p.Name), lowerName) or string.find(string.lower(p.DisplayName), lowerName) then return p end
    end
    return nil
end

BringBtn.MouseButton1Click:Connect(function()
    bringTarget = getPlayer(BringBox.Text)
    if not bringTarget then return end
    blackHoleActive = not blackHoleActive
    if blackHoleActive then
        BringBtn.Text = "Bring Parts | On"; BringBtn.BackgroundColor3 = Color3.fromRGB(0, 180, 0)
        for _, v in ipairs(Workspace:GetDescendants()) do ForcePart(v) end
        DescendantAddedConnection = Workspace.DescendantAdded:Connect(function(v) if blackHoleActive then ForcePart(v) end end)
    else
        BringBtn.Text = "Bring Parts | Off"; BringBtn.BackgroundColor3 = Color3.fromRGB(0, 120, 215)
        if DescendantAddedConnection then DescendantAddedConnection:Disconnect() end
    end
end)

-- Render Loop chính
RunService.Heartbeat:Connect(function()
    local char = LocalPlayer.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if root then
        PosDisplay.Text = string.format("X:%.0f Y:%.0f Z:%.0f", root.Position.X, root.Position.Y, root.Position.Z)
        
        -- Follow Logic
        if isFollowing and targetPlayer and targetPlayer.Character then
            local tRoot = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
            if tRoot then
                for _, v in pairs(char:GetDescendants()) do if v:IsA("BasePart") then v.CanCollide = false end end
                local dist = tonumber(DistInput.Text) or 7
                local tPos = (tRoot.CFrame * CFrame.new(0, 1.5, dist)).Position
                root.CFrame = CFrame.lookAt(tPos, Vector3.new(tRoot.Position.X, tPos.Y, tRoot.Position.Z))
                root.Velocity = Vector3.new(0,0,0)
            end
        end

        -- Bring Parts Logic
        if blackHoleActive and bringTarget and bringTarget.Character then
            local tRoot = bringTarget.Character:FindFirstChild("HumanoidRootPart")
            if tRoot then Attachment1.WorldCFrame = tRoot.CFrame end
        end
    end
end)
